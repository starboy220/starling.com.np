Advanced Java Concepts

Generics

// Generic class
class Box<T> {
    private T content;
    
    public void set(T content) {
        this.content = content;
    }
    
    public T get() {
        return content;
    }
}

// Usage
Box<String> stringBox = new Box<>();
stringBox.set("Hello Generics!");

Lambda Expressions

// Functional interface
interface Greeting {
    void greet(String name);
}

// Using lambda
Greeting g = (name) -> {
    System.out.println("Hello, " + name);
};

g.greet("Alice");
