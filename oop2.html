Object-Oriented Programming

Classes and Objects

public class Car {
    // Fields
    String color;
    String model;
    
    // Method
    void startEngine() {
        System.out.println("Engine started!");
    }
}

// Creating an object
Car myCar = new Car();
myCar.color = "Red";
myCar.startEngine();

Inheritance

class Animal {
    void eat() {
        System.out.println("Eating...");
    }
}

class Dog extends Animal {
    void bark() {
        System.out.println("Barking...");
    }
}

Dog myDog = new Dog();
myDog.eat();  // Inherited method
myDog.bark(); // Own method
